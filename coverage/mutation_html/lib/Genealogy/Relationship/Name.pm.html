  459:     name ______________________________________________________
  460:     [In]  steps_to_ancestor   : N0

  462:           sex                 : {M, F}
  463:           language            : {en, fr, de}?  (default en)
  464:           person              : Object?
  465:     [Out] result              : String | undef
  466: 
  467:     Let key == steps_to_ancestor ++ "," ++ steps_from_ancestor

  469:     result == table(key)  if key in dom table
  470:            == undef       otherwise

  475: 	my $self = shift;
  476: 
  477: 	# Validate and extract the remaining parameters; capture the return value
  478: 	my $args = Params::Validate::Strict::validate_strict(
  479: 		args   => Params::Get::get_params(undef, \@_) || {},
  480: 		schema => {
  481: 			steps_to_ancestor   => { type => 'integer', minimum => 0 },
  482: 			steps_from_ancestor => { type => 'integer', minimum => 0 },
  483: 			sex                 => { type => 'string', memberof => ['M', 'F'] },
  484: 			language            => { type => 'string', regex => qr/^(?:en|de|fr)/,
  485: 			                         optional => 1 },
  486: 			person              => { type => 'object', optional => 1 },
  487: 		}
  488: 	);
  489: 
  490: 	# Extract individual parameters; undef means arg was not given, so croak
  491: 	my $steps1 = $args->{steps_to_ancestor} // croak('steps_to_ancestor not given');
  492: 	my $steps2 = $args->{steps_from_ancestor} // croak('steps_from_ancestor not given');
  493: 	my $sex    = $args->{sex} // croak('sex not given');
  494: 	my $person = $args->{person};
  495: 
  496: 	# Fall back to constructor default or hard default if no per-call language given
  497: 	my $lang = lc($args->{language} // $self->{language} // $DEFAULT_LANGUAGE);
  498: 
  499: 	# Strip any region subtag (e.g. 'en-GB' -> 'en') after lowercasing
  500: 	($lang) = split /-/, $lang;
  501: 
  502: 	# Build lookup key from the two step counts
  503: 	my $key = "${steps1},${steps2}";
  504: 
  505: 	# Retrieve the correct gender-specific table for the chosen language
  506: 	my $table = $RELATIONSHIP_TABLES{$lang}{$sex};
  507: 
  508: 	# Look up and return; undef when the combination is not tabulated
  509: 	my $result = $table->{$key};
  510: 
  511: 	return $result;
  512: }

  642: =head2 Error handling
  643: 
  644: Errors are dispatched through the following priority chain:
  645: 
  646: =over 4
  647: 
  648: =item 1. L<Log::Abstraction> object (preferred)
  649: 
  650: Construct a C<Log::Abstraction> object with the desired logger coderef and
  651: C<ctx> (typically a C<Gedcom::Individual>), then pass it as C<logger> to
  652: C<new()>.  On error, this module simply calls C<< $logger->error($msg) >>
  653: and Log::Abstraction handles ctx forwarding, formatting, and dispatch.
  654: 
  655:     use Log::Abstraction;
  656: 
  657:     my $logger = Log::Abstraction->new(
  658:         logger => sub {
  659:             my $args = shift;
  660:             my $msg = $args->{ctx}
  661:                 ? $args->{ctx}->as_string() . ': ' . join('', @{$args->{message}})
  662:                 : join('', @{$args->{message}});
  663:             complain({ message => $msg, person => $args->{ctx} });
  664:         },
  665:         ctx => $individual,
  666:     );
  667: 
  668:     my $namer = Genealogy::Relationship::Name->new(logger => $logger);
  669: 
  670: =item 2. C<on_error> coderef
  671: 
  672: A simpler fallback for callers not using L<Log::Abstraction>.  The coderef
  673: is called with a flat hash matching the C<complain()> interface in
  674: C<gedcom>/C<ged2site>: C<warning =E<gt> $msg, person =E<gt> $person>.
  675: C<person> is resolved from the per-call C<name()> C<person> argument first,
  676: then from the constructor C<ctx>.
  677: 
  678:     my $namer = Genealogy::Relationship::Name->new(
  679:         on_error => sub {
  680:             my %args = @_;
  681:             complain(warning => $args{warning}, person => $args{person});
  682:         },
  683:     );
  684: 
  685: =item 3. C<croak> (default)
  686: 
  687: When neither C<logger> nor C<on_error> is set, errors croak normally.
  688: 
  689: =back
  690: 
  691: If any handler returns without dying (e.g. a warning-only handler in
  692: L<Log::Abstraction>), C<name()> returns C<undef>.
